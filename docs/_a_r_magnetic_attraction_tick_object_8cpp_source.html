26void UARMagneticAttractionTickObject::OnTick(const FARPhysicsTickParameters& TickParams, FARPhysicsEvaluationResult& Result)
27{
28 const AActor* targetActor = Target->GetActor();
29 if (targetActor == nullptr)
30 {
31 AR_LOG(LogARPhysics, Error, TEXT("Target actor is nullptr.Do you override IARMagnetizableInterface::GetActor()? "));
32 return;
33 }
34
35 // 斥力計算
36 for (const auto& magnetizedObject : AffectedMagnetizedObjects)
37 {
38 if ((magnetizedObject == nullptr) || (magnetizedObject->GetActor() == nullptr))
39 {
40 continue;
41 }
42
43 // TODO Maybe should merge this things into EngineProxy?
44 {
45 const AActor* magnetizedActor = magnetizedObject->GetActor();
46 const FVector directionTo = magnetizedActor->GetActorLocation() - targetActor->GetActorLocation();
47 const FVector pushForce = directionTo.GetUnsafeNormal() * PROPORTIONALITY_CONSTANT * MAGNETIC_VALUE * MAGNETIC_VALUE / directionTo.SizeSquared() * 0.0001f;
48
49 Result.ForceResult += pushForce;
50
51 }
52 }
53}
