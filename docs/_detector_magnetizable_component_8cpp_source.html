52void UDetectorMagnetizableComponent::UpdateDetectedActors()
53{
54 if (m_OwnerActor == nullptr) { return; }
55
56 TArray<AActor*> overlappedActors;
57 TArray<AActor*> detectedActors;
58
59 // プレイヤーを中心とした球体範囲内のアクターを取得
60 UKismetSystemLibrary::SphereOverlapActors(
61 GetWorld(),
62 m_OwnerActor->GetActorLocation(), /*中心座標*/
63 m_DetectionRadius, /*検知範囲*/
64 m_ObjectTypes, /*検知するオブジェクトタイプ(UE側で指定)*/
65 nullptr, /*すべてのクラスを検知するクラス*/
66 { m_OwnerActor }, /*無視するアクター*/
67 overlappedActors /*検知したアクターの格納場所*/
68 );
69
70 // IARMagnetizableInterface 実装しているアクターを抽出
71 for (AActor* Actor : overlappedActors)
72 {
73 if (Actor == nullptr) { continue; }
74
75 // クラスフィルタに合うか判定
76 for (TSubclassOf<AActor> AllowedClass : m_DetectionClassFilter)
77 {
78 if (Actor->IsA(AllowedClass))
79 {
80 if (!detectedActors.Contains(Actor))
81 {
82 detectedActors.Add(Actor);
83 }
84 }
85 }
86 }
87
88 // 検知したものが変わった場合処理
89 if (detectedActors != m_DetectedMagnetizableActors)
90 {
91 // 新しく検知したオブジェクトへの処理
92 for (AActor* Actor : detectedActors)
93 {
94 if (!m_DetectedMagnetizableActors.Contains(Actor))
95 {
96 m_DetectedMagnetizableActors.Add(Actor);
97 SetActorOnOutline.ExecuteIfBound(Actor);
98 }
99 }
100
101 TArray<AActor*> RemoveActors;
102 // 検知しなくなったオブジェクトへの処理
103 for (AActor* Actor : m_DetectedMagnetizableActors)
104 {
105 if (!detectedActors.Contains(Actor))
106 {
107 RemoveActors.Add(Actor);
108 UnsetActorOnOutline.ExecuteIfBound(Actor);
109 }
110 }
111 for (AActor* Actor : RemoveActors)
112 {
113 m_DetectedMagnetizableActors.Remove(Actor);
114 }
115 }
116}

121void UDetectorMagnetizableComponent::AssignTargetMagnetizableObject()
122{
123 // 変数宣言
124 AActor* currentTarget;
125 FVector startPoint;
126 FVector endPoint;
127
128 // カメラが有効でなければリターン
129 if (m_PlayerCameraComponent == nullptr || !m_PlayerCameraComponent->GetEvaluationContext().IsValid()) { return; }
130
131 // 始点と終点を計算
132 startPoint = m_OwnerActor->GetActorLocation(); /*プレイヤーの座標*/
133 endPoint = startPoint + GetPlayerCameraRotation() * m_DetectionRadius; /*付与できる範囲*/
134
135 // ライントレースで取得
136 currentTarget = TraceForMagnetizableObject(startPoint, endPoint);
137
138 // 対象が変更された場合
139 if (currentTarget != m_TargetMagnetizableActor)
140 {
141 // 前の対象を解除
142 if (m_TargetMagnetizableActor != nullptr && m_TargetMagnetizableActor->GetClass()->ImplementsInterface(UARMagnetizableInterface::StaticClass()))
143 {
144 UnsetMagnetizableObjectAtCursor.ExecuteIfBound(m_TargetMagnetizableActor);
145 }
146
147 // 新しい対象を設定
148 if (currentTarget != nullptr && currentTarget->GetClass()->ImplementsInterface(UARMagnetizableInterface::StaticClass()))
149 {
150 SetMagnetizableObjectAtCursor.ExecuteIfBound(currentTarget);
151 m_TargetMagnetizableActor = currentTarget;
152 }
153 else
154 {
155 m_TargetMagnetizableActor = nullptr;
156 }
157 }
158}

203AActor* UDetectorMagnetizableComponent::TraceForMagnetizableObject(const FVector& startpoint, const FVector& endpoint)
204{
205 FHitResult HitResult;
206 FCollisionQueryParams Params;
207 Params.AddIgnoredActor(m_OwnerActor);
208 Params.bReturnPhysicalMaterial = false;
209
210 const bool bHit = GetWorld()->LineTraceSingleByChannel(
211 HitResult,
212 startpoint,
213 endpoint,
214 ECC_Visibility,
215 Params
216 );
217
218 if (bHit && HitResult.GetActor() != nullptr)
219 {
220 return HitResult.GetActor();
221 }
222
223 return nullptr;
224}
