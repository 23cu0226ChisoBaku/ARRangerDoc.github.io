54void AInsekiGameMode::BeginPlay()
55{
56 Super::BeginPlay();
57
58 GameResultTimerHandle.Invalidate();
59 // Register GameInstance OnReset
60 if (UARGameInstance* ARGI = ::Cast<UARGameInstance>(GetGameInstance()))
61 {
62 ARGI->OnReset.AddUObject(this, &ThisClass::OnResetCommandSent);
63 }
64
65 // 音声データを初期化
66 const UWorld* world = GetWorld();
67 if (world != nullptr)
68 {
69 UARAudioSystem* audioSystem = world->GetGameInstance()->GetSubsystem<UARAudioSystem>();
70 if (audioSystem != nullptr)
71 {
72 audioSystem->InitializeSounds( nullptr, SoundEffectData);
73 }
74 }
75
76 // BlinkingOutlineWorldSubsystem を取得
77 UBlinkingOutlineWorldSubsystem* WorldSubsystem = GetWorld()->GetSubsystem<UBlinkingOutlineWorldSubsystem>();
78 if (WorldSubsystem != nullptr)
79 {
80 APlayerController* PC = GetWorld()->GetFirstPlayerController();
81 if (PC != nullptr && PC->GetPawn() != nullptr)
82 {
83 // プレイヤーのPawnからLineTraceSingleARObjectComponentを取得
84 UDetectorMagnetizableComponent* DetectorMagnetizableComp = PC->GetPawn()->FindComponentByClass<UDetectorMagnetizableComponent>();
85
86 // TickActorClassを設定し、サブシステムにセットアップを依頼
87 WorldSubsystem->SetupBlinkingSystem(GetWorld(), DetectorMagnetizableComp, OutlineTickActorClass);
88 }
89 }
90
92 InitializeObserver();
93 InitializeEvents();
94 InitializeOnMapEnemies();
95 InitializeARPhysics();
96
97 // Register debug key
98 ARRanger::Global::RegisterDebugKey();
99
100 /*Event Manager デリゲートバインド*/
101 RegisterBattleEventDelegate();
102
103 SetGameUserSettings(gSettingsDataStack);
104}

279void AInsekiGameMode::ProcessGameResult(EGameResultState ResultState)
280{
281 if (bGameResultHandled)
282 {
283 return;
284 }
285
286 // Set clear timer
287 auto gameResultHandler = [this, ResultState]()
288 {
289 if (UARGameInstance* gameInst = this->GetGameInstance<UARGameInstance>())
290 {
291 switch (ResultState)
292 {
293 case GameClear:
294 {
295 gameInst->ProcessGameClear();
296 }
297 break;
298
299 case GameOver:
300 {
301 gameInst->ProcessGameOver();
302 }
303 break;
304 }
305 }
306 };
307
308 // TODO
309 GetWorldTimerManager().SetTimer(GameResultTimerHandle, gameResultHandler, 6.0f, false);
310 bGameResultHandled = true;
311}
