29void ABattleEventField::BeginPlay()
30{
31 Super::BeginPlay();
32
33 /*Tick処理を無効化*/
34 SetActorTickEnabled(false);
35
36 /*バトルフィールド範囲用（コリジョン）コンポーネントを全て取得*/
37 GetPrimitiveComponents();
38
39 /*各範囲コリンジョンコンポーネントににオーバーラップ関数をバインド*/
40 for (UPrimitiveComponent* primComp : m_PrimitiveComponents)
41 {
42 if (primComp == nullptr)
43 {
44 continue;
45 }
46 primComp->OnComponentBeginOverlap.AddDynamic(this, &ABattleEventField::OnFieldBeginOverlap);
47 }
48
49 /*バリアエフェクトのコンポーネントを取得する*/
50 if(m_BarrierEffectActor != nullptr)
51 {
52 UNiagaraComponent* niagaraComp = m_BarrierEffectActor->GetNiagaraComponent();
53 if (niagaraComp)
54 {
55 m_BarrierEffectComp = niagaraComp;
56 /*最初はナイアガラを再生しない*/
57 m_BarrierEffectComp->bAutoActivate = false;
58 }
59 }
60
61 /*次フレームで範囲内のスポナーを取得*/
62 GetWorld()->GetTimerManager().SetTimerForNextTick(this, &ABattleEventField::CollectSpawners);
63}

65void ABattleEventField::Tick(float deltaTime)
66{
67 Super::Tick(deltaTime);
68
69 if (m_Player == nullptr)
70 {
71 return;
72 }
73
74 /*どれか一つのフィールドコリジョンにプレイヤーが完全に入っているか*/
75 FBox playerBox = m_Player->GetComponentsBoundingBox();
76 for (UPrimitiveComponent* primComp : m_PrimitiveComponents)
77 {
78 if (primComp == nullptr)

80 continue;

84

118void ABattleEventField::CollectSpawners()
119{
120 m_Spawners.Empty();
121 m_RemainingEnemiesInField = 0;
122 m_FieldPhases.Empty();
123
124 /*自身のUPrimitiveComponentを全て取得*/
125 for (UPrimitiveComponent* primComp : m_PrimitiveComponents)
126 {
127 if (primComp == nullptr)
128 {
129 continue;
130 }
131
132 /*UPrimitiveComponent と Overlap しているスポナーを全て取得*/
133 TArray<AActor*> overlappingSpawners;
134 primComp->GetOverlappingActors(overlappingSpawners, AEnemySpawner::StaticClass());
135 for (AActor* actor : overlappingSpawners)
136 {
137 if (AEnemySpawner* spawner = Cast<AEnemySpawner>(actor))
138 {
139 m_Spawners.Add(spawner);
140
141 if (GEngine) { GEngine->AddOnScreenDebugMessage(-1, 5.f, FColor::Blue, TEXT("aaaaaaaaaaaa")); }
142 /*スポナーが持つフェーズを収集*/
143 const TSet<ESpawnPhase> phases = spawner->GetSpawnPhases();
144 for (const ESpawnPhase& phase : phases)
145 {
146 /*フィールドに湧く敵の数をカウント*/
147 ++m_RemainingEnemiesInField;
148
149 /*すでにあれば入れない*/
150 if (!m_FieldPhases.Contains(phase))
151 {
152 m_FieldPhases.Add(phase);
153 }
154 }
155 }
156 }
157
158 /*ESpawnPhase の定義順にソート*/
159 m_FieldPhases.Sort([](const ESpawnPhase& A, const ESpawnPhase& B) {
160 return static_cast<uint8>(A) < static_cast<uint8>(B);
161 });
162 }
163
164 /*スポナーの取得完了通知*/
165 OnSpawnersCollected.Broadcast(this);
166}

251void ABattleEventField::RequestSpawn()
252{
253 for (TObjectPtr<AEnemySpawner> spawner : m_Spawners)
254 {
255 if (spawner == nullptr)
256 {
257 continue;
258 }
259
260 /*スポナーが対象フェーズを持っていれば敵をスポーン*/
261 if (m_FieldPhases.IsValidIndex(m_CurrentPhaseIndex - 1))
262 {
263 ESpawnPhase CurrentPhase = m_FieldPhases[m_CurrentPhaseIndex- 1];
264 if (spawner->GetSpawnPhases().Contains(CurrentPhase))
265 {
266 /*スポナーに敵の生成を促し、敵死亡時のコールバック関数を引数で渡す*/
267 FScriptDelegate delegate;
268 delegate.BindUFunction(this, FName("OnEnemyDestroyed"));
269 spawner->SpawnEnemy(delegate);
270
271 /*敵の数をカウント*/
272 ++m_RemainingEnemiesInPhase;
273 }
274 }
275 }

291void ABattleEventField::OnEnemyDestroyed()
292{
293 /*残りの敵キャラクターの数を減らす*/
294 --m_RemainingEnemiesInField;
295 --m_RemainingEnemiesInPhase;
296
297 if (m_RemainingEnemiesInPhase <= 0)
298 {
299 /*妥協処理*/
300 EventEndingPhase(m_CurrentPhaseIndex);
301
302 /*次のフェーズがある*/
303 if (m_CurrentPhaseIndex < m_FieldPhases.Num())
304 {
305 /*現在のフェーズ数を更新*/
306 ++m_CurrentPhaseIndex;
307 /*スポーン開始*/
308 StartNextPhase();
309 }
310 /*フィールド内の敵を全滅させた*/
311 else
312 {
313 /*バトルイベント終了処理*/
314 OnEndBattleEvent();
315 }
316 }
