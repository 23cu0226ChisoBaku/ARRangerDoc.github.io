57void UARPlayerInputBuffer::EvaluateBuffer(const float DeltaTime, const bool bGamePaused)
58{
59 // Skip evaluation if bEvaluateIfGamePaused is false and bGamePaused is true
60 if (!bEvaluateIfGamePaused && bGamePaused)
61 {
62 return;
63 }
64
65 for (const auto& inputState : m_inputStates)
66 {
67 if (inputState.IsValid())

69 inputState->Evaluate(PlayerController, DeltaTime);

71 }
72

86void UARPlayerInputBuffer::HandleInputTagPressed(const FGameplayTag& InInputTag, bool bOverrideInputState)
87{
88 if (InputKeepTime > 0.0f)
89 {
90 bool bGenerateNew = true;
91 for (const TPimplPtr<ARRanger::Input::FARInputBufferState>& inputState : m_inputStates)
92 {
93 if (inputState.IsValid() && inputState->IsInputTagMatchesExact(InInputTag))
94 {
95 bGenerateNew = false;
96 // Refresh buffer during input pressed
97 if (bOverrideInputState || inputState->IsPressedState())
98 {
99 inputState->MarkAsPressed();
100 inputState->SetLifeTime(InputKeepTime);
101 break;
102 }
103 }
104 }
105
106 if (bGenerateNew)
107 {
108 m_inputStates.Emplace(ARRanger::Input::FARInputBufferState::MakePressedState(InInputTag, InputKeepTime));
109 }
110 }
111}

139void UARPlayerInputBuffer::ConsumeBuffer(const FGameplayTag& InInputTag)
140{

142 {
143 if (!inputState.IsValid())
144 {
145 continue;
146 }
147

155void UARPlayerInputBuffer::RemoveInputBufferInternal()
156{
157 int32 idx = 0;
158 while (idx < m_inputStates.Num())
159 {
160 const TPimplPtr<ARRanger::Input::FARInputBufferState>& inputState = m_inputStates[idx];
161 // Remove expired buffer
162 if (!inputState.IsValid() || inputState->IsInputStateExpired())
163 {
164 m_inputStates.RemoveAt(idx);
165 continue;
166 }
167
168 ++idx;
169 }
170}
