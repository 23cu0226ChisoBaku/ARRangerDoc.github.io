59void UARGameplayAbility_Jump::CharacterJumpStart()
60{
61 if (ACharacter* character = ::Cast<ACharacter>(GetAvatarActorFromActorInfo()))
62 {
63 if (character->IsLocallyControlled() && !character->bPressedJump)
64 {
65 character->UnCrouch();
66 character->Jump();
67
68 character->LandedDelegate.AddUniqueDynamic(this, &ThisClass::OnCharacterLanded);
69 }
70 }
71}
72
73void UARGameplayAbility_Jump::CharacterJumpEnd()
74{
75 if (ACharacter* character = ::Cast<ACharacter>(GetAvatarActorFromActorInfo()))
76 {
77 if (character->IsLocallyControlled() && character->bPressedJump)
78 {
79 character->StopJumping();
80 character->LandedDelegate.RemoveDynamic(this, &ThisClass::OnCharacterLanded);
81 }
82 }
83}
