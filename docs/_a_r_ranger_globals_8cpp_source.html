21 bool HandleKeyDownEvent(FSlateApplication& SlateApp, const FKeyEvent& InKeyEvent) override
22 {
23 // Ignore Key that is combine with modifier key
24 if (InKeyEvent.IsAltDown() || InKeyEvent.IsCommandDown() || InKeyEvent.IsControlDown() || InKeyEvent.IsShiftDown())
25 {
26 return false;
27 }
28
29 bool bKeyDownHandled = false;
30 const FKey pressedKey = InKeyEvent.GetKey();
31 // Reset Game button
32 if (pressedKey == EKeys::F1)
33 {
34 if (m_gameInst.IsValid())
35 {
36 m_gameInst->ResetGame();
37 bKeyDownHandled = true;
38 }
39 }
40 else if (pressedKey == EKeys::F11)
41 {
42 if (m_gameInst.IsValid())
43 {
44 m_gameInst->ResetBattleTestStage();
45 bKeyDownHandled = true;
46 }
47 }
48
49 // For Demo Build
50#if AR_DEMO_BUILD
51
52 const FName PunchLevelName = FName{ "Demo_Punch_Stage" };
53 const FName SpecialLevelName = FName{ "Demo_ChargeKick_Special_Stage" };
54 const FName AttractionLevelName = FName{ "Demo_AttractionClimb_Stage" };
55 const FName RepulsionLevelName = FName{ "Demo_RepulsionJump_Stage" };
56
57 if (pressedKey == EKeys::F5)
58 {
59 UGameplayStatics::OpenLevel(m_gameInst.Get(), PunchLevelName);
60 bKeyDownHandled = true;
61 }
62 else if (pressedKey == EKeys::F6)
63 {
64 UGameplayStatics::OpenLevel(m_gameInst.Get(), SpecialLevelName);
65 bKeyDownHandled = true;
66 }
67 else if (pressedKey == EKeys::F7)
68 {
69 UGameplayStatics::OpenLevel(m_gameInst.Get(), AttractionLevelName);
70 bKeyDownHandled = true;
71 }
72 else if (pressedKey == EKeys::F8)
73 {
74 UGameplayStatics::OpenLevel(m_gameInst.Get(), RepulsionLevelName);
75 bKeyDownHandled = true;
76 }
77#endif
78
79 return bKeyDownHandled;
80 }

99 void RegisterDebugKey()

102 {
103 for (const FWorldContext& worldCtx : GEngine->GetWorldContexts())
104 {
105 UGameInstance* gameInst = worldCtx.OwningGameInstance;
106 if (UARGameInstance* arGameInst = ::Cast<UARGameInstance>(gameInst))
107 {
108 if (s_debugInputProcessor.IsValid())
109 {
110 UnregisterDebugKey();

112
113 s_debugInputProcessor = FARDebugInputProcessor::MakeInstance(arGameInst);

115 }
116 }
117 }
