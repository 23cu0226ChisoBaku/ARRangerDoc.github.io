47void UAnimNotifyState_DetectRange::NotifyBegin(USkeletalMeshComponent * MeshComp, UAnimSequenceBase * Animation, float TotalDuration, const FAnimNotifyEventReference& EventReference)
48{
49 Super::NotifyBegin(MeshComp, Animation, TotalDuration, EventReference);
50
51 if (ValidateParameters(MeshComp))
52 {
53 if (RangeData != nullptr)
54 {
55 // TODO
56 FDetectorAssetEntry DAE{};
57 DAE.DetectorData = RangeData;
58 DAE.Target.TargetType = EDetectorTargetType::Interface;
59 DAE.Target.TargetInterface = UARAttackable::StaticClass();
60 DAE.Priority = 0;
61
62 URangeDetectorComponent* RDC = URangeDetectorHelper::AttachRangeDetector(DAE, MeshComp, SocketName, LocationOffset, RotationOffset, FVector::OneVector, EAttachLocation::KeepRelativeOffset, true);
63
64
65 if (!m_detectTickObject.IsValid() && DetectionType == EANS_DetectRange_NotifyDetectionType::NotifyDuringHit)
66 {
67 using enum EANS_DetectRange_NotifyDetectionFrequencyType;
68 switch (FrequencyType)
69 {
70 case Notify_FrameBase:
71 {
72 // TODO Currently we do not use startFrame
73 const int32 startFrame = 0;
74 m_detectTickObject = ::MakeUnique<FDetectTickObject_FrameBase>(startFrame, FrameInterval);
75 }
76 break;
77
78 case Notify_TimeBase:
79 {
80 m_detectTickObject = ::MakeUnique<FDetectTickObject_TimeBase>(TimeInterval);
81 }
82 break;
83 }
84
85 if (m_detectTickObject.IsValid())
86 {
87 m_detectTickObject->DetectionNotify.BindUObject(this, &UAnimNotifyState_DetectRange::NotifyAbility);
88 }
89 }
90
91#if WITH_EDITORONLY_DATA
92
93 if (RDC != nullptr)
94 {
95 RDC->bDrawDebugRange = bDrawDebugDuringActivation;
96 }
97
98#endif // WITH_EDITORONLY_DATA
99 }
100 }
101
102 m_notifiedActors.Reset();
103}

133void UAnimNotifyState_DetectRange::NotifyEnd(USkeletalMeshComponent * MeshComp, UAnimSequenceBase * Animation, const FAnimNotifyEventReference& EventReference)
134{
135 TArray<USceneComponent*> children{};
136 MeshComp->GetChildrenComponents(false, children);
137
138 for (USceneComponent* child : children)
139 {
140 URangeDetectorComponent* RDC = ::Cast<URangeDetectorComponent>(child);
141 if (RDC != nullptr)
142 {
143 const bool bSocketMatch = RDC->GetAttachSocketName() == SocketName;
144 const bool bRangeDataMatch = RDC->HasRangeData(RangeData);
145
146 if (bSocketMatch && bRangeDataMatch)
147 {
148 RDC->RemoveDetector(RangeData);
149
150 if (bDestroyAtEnd && RDC->IsDetectorEmpty())
151 {
152 RDC->DestroyComponent();
153 }
154 else
155 {
156 RDC->DetachFromComponent(FDetachmentTransformRules::KeepWorldTransform);
157 }
158
159 break;
160 }
161 }
162 }
163
164 Super::NotifyEnd(MeshComp, Animation, EventReference);
165
166 m_notifiedActors.Reset();
167
168#if WITH_EDITOR
169
170 if (MeshComp->GetWorld() != nullptr)
171 {
172 if (bDrawDebugDuringActivation)
173 {
174 ULineBatchComponent* lineBatch = MeshComp->GetWorld()->GetLineBatcher(UWorld::ELineBatcherType::WorldPersistent);
175
176 // Flush all debug lines drawn in NotifyTick
177 if (lineBatch != nullptr)
178 {
179 lineBatch->Flush();
180 }
181 }
182 }
183
184#endif
185}

213void UAnimNotifyState_DetectRange::NotifyAbility(USkeletalMeshComponent* MeshComp, UAnimSequenceBase * Animation)
214{
215 if (MeshComp == nullptr)
216 {
217 return;
218 }
219
220 TArray<USceneComponent*> children{};
221 MeshComp->GetChildrenComponents(false, children);
222
223 for (USceneComponent* child : children)
224 {
225 URangeDetectorComponent* RDC = ::Cast<URangeDetectorComponent>(child);
226 if ((RDC == nullptr) || !RDC->HasRangeData(RangeData))
227 {
228 continue;
229 }
230
231 FRangeDetectorEvaluationResult result{};
232 const int32 resultCnt = RDC->GetResultByRangeData(RangeData, result);
233 if (resultCnt <= 0)
234 {
235 continue;
236 }
237
238 // Notify Ability
239 UARAbilitySystemComponent* ARASC = UARAbilitySystemComponent::FindARAbilitySystemComponent(MeshComp->GetOwner());
240 if (ARASC != nullptr)
241 {
242 const bool bIsOnlyNotifyOnce = DetectionType == EANS_DetectRange_NotifyDetectionType::NotifyOnce;
243
244 const TArray<FGameplayAbilitySpec>& allActivatableAbilites = ARASC->GetActivatableAbilities();
245 for (const FGameplayAbilitySpec& activatableAbility : allActivatableAbilites)
246 {
247 if (activatableAbility.IsActive())
248 {
249 if (IARGameplayAbilityNotifyInterface* notifyInterface = ::Cast<IARGameplayAbilityNotifyInterface>(activatableAbility.GetPrimaryInstance()))
250 {
251 TArray<FGANotify_ImpactResult> impactResults{};
252
253 for (const auto& detectedActor : result.DetectedActors)
254 {
255 // TODO Only for notify once use. Maybe we should need some other solution
256 if (bIsOnlyNotifyOnce && m_notifiedActors.Contains(detectedActor))
257 {
258 continue;
259 }
260
261 m_notifiedActors.Emplace(detectedActor);
262
263 FGANotify_ImpactResult impactResult{
264 .SourceActor = MeshComp->GetOwner(),
265 .HitActor = detectedActor,
266 .OccurrenceComp = RDC,
267 .ImpactLocation = RDC->GetComponentLocation()
268 };
269
270 impactResults.Emplace(impactResult);
271 }
272
273 notifyInterface->GANotify_ImpactResult(MeshComp, Animation, impactResults);
274 }
275 }
276 }
277 }
278 }
