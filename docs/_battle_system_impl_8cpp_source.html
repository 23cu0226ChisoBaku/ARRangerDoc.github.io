46 // Attackerが存在しないため、Attackerへの通知を送らない

50 }
51
53 // Use Instigator if Instigator is valid and bUseAttackerActor is false, otherwise use actor of IARAttackerInterface
54 const bool bUseIntigator = (InAttackParams.Instigator != nullptr) && InAttackParams.bUseAttackerActor;
55 if (!bUseIntigator)
56 {
57 InAttackParams.Instigator = Attacker != nullptr ? Attacker->GetActor() : nullptr;
58 }
59
60 FARAttackResult outAttackResult{};
61 OnPreAttacked(InAttackParams, outAttackResult);
62
64 if (Attacker != nullptr)
65 {
66 FARAttackNotifyParameter notifyParams{};
67 notifyParams.WeakAttackableObject = _getUObject();
68 Attacker->NotifyAttackResult(outAttackResult.Result, notifyParams);
69 }
70
72 OnPostAttacked(InAttackParams);
73
75 // 攻撃が失敗したらバトルタスクを処理しない
76 if (outAttackResult.Result != EARAttackResult::Success)
77 {
78 return false;
79 }
80
81 // バトルタスクを処理する
82 FARBattleTask task{};
83 task.Instigator = InAttackParams.Instigator;
84 task.Target = this->Attackable_GetActor();
85 task.OriginDamage = InAttackParams.Damage;
86
87 FARDamageResult damageResult{};
88 IBattleSystemInterface::Get().HandleBattleTask(task, damageResult);
89
90 // ダメージ通知
91 damageResult.FinalLaunchDirection = InAttackParams.LaunchDirection;
92 damageResult.Instigator = (Attacker != nullptr) ? Attacker->GetActor() : nullptr;
93 damageResult.ImpactLocation = InAttackParams.ImpactLocation;
94 OnDamaged(damageResult);
95
96 // Attack success
97 return true;
