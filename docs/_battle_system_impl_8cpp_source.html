31bool IARAttackable::AttackTarget(IARAttackerInterface* Attacker, FARAttackParameters InAttackParams)
32{
33 // Attackerが存在しないため、Attackerへの通知を送らない
34 if (Attacker == nullptr)
35 {
36 AR_LOG(LogARBattle, Error, TEXT("Attacker is INVALID!!!"));
37 }
38
40 // Use Instigator if Instigator is valid and bUseAttackerActor is false, otherwise use actor of IARAttackerInterface
41 const bool bUseIntigator = (InAttackParams.Instigator != nullptr) && InAttackParams.bUseAttackerActor;
42 if (!bUseIntigator)
43 {
44 InAttackParams.Instigator = Attacker != nullptr ? Attacker->GetActor() : nullptr;
45 }
46
47 // Check attack result
48 ARRanger::Battle::FARAttackResult outAttackResult{};
49 OnPreAttacked(InAttackParams, outAttackResult);
50
52 if (Attacker != nullptr)
53 {
54 ARRanger::Battle::FARAttackNotifyParameter notifyParams{};
55 notifyParams.WeakAttackableObject = _getUObject();
56 Attacker->NotifyAttackResult(outAttackResult.Result, notifyParams);
57 }
58
60 OnPostAttacked(InAttackParams);
61
63 // Do not handle battle task if attack is not success
64 if (outAttackResult.Result != ARRanger::Battle::EARAttackResult::Success)
65 {
66 return false;
67 }
68
69 // Finally we handle battle task
70 ARRanger::Battle::IBattleSystemInterface& battleSystem = ARRanger::Battle::IBattleSystemInterface::Get();
71 ARRanger::Battle::FARBattleTask task{};
72 ARRanger::Battle::FARDamageResult damageResult{};
73 task.Instigator = InAttackParams.Instigator;
74 task.Target = this->Attackable_GetActor();
75 task.OriginDamage = InAttackParams.Damage;
76 battleSystem.HandleBattleTask(task, damageResult);
77
78 // Handle damage
79 damageResult.FinalLaunchDirection = InAttackParams.LaunchDirection;
80 damageResult.Instigator = (Attacker != nullptr) ? Attacker->GetActor() : nullptr;
81 damageResult.ImpactLocation = InAttackParams.ImpactLocation;
82 OnDamaged(damageResult);
83
84 // Attack success
85 return true;
86}
