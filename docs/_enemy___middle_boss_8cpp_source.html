73void AEnemy_MiddleBoss::OnAttackPerformed(EAttackType InAttackType)
74{
75 K2_OnAttackPerformed(InAttackType);
76
77 Task.Reset();
78
79 switch (InAttackType)
80 {
81 case EAttackType::Punch:

83 Task = ::MakeUnique<FAttackTask_ApproachTarget>(this, TargetActor, EAttackType::Punch, PunchRange);

85 break;
86
87 case EAttackType::JumpAttack:

89 Task = ::MakeUnique<FAttackTask_ApproachTarget>(this, TargetActor, EAttackType::JumpAttack, JumpAttackRange);

92 }
93 break;
94
95 case EAttackType::Slammed:
96 {
97 Task = ::MakeUnique<FAttackTask_PreSlammed>(this);
98 }
99 break;
100
101 case EAttackType::Roar:
102 {
103 Task = ::MakeUnique<FAttackTask_PreRoar>(this);
104 }

217void AEnemy_MiddleBoss::ReceiveLaunch(const FVector& LaunchDirection)
218{
219 bDeadLaunchHandled = true;
220
221 // 死亡時は大きく吹っ飛ばす
222 const float launchPower = 10000.0f;
223 FVector DeathImpulse = LaunchDirection;
224
225 TArray<AActor*> actors{};
226 UGameplayStatics::GetAllActorsWithTag(this, FName{"Landmark"}, actors);
227 if (actors.Num() > 0)
228 {
229 if (actors[0] != nullptr)
230 {
231 const FVector relativeOffsetToLogo = FVector{0.0, 80.0, 400.0};
232 const FVector pointToHitLandmark = actors[0]->GetActorLocation() + relativeOffsetToLogo * actors[0]->GetActorScale3D();
233 DeathImpulse = (pointToHitLandmark - GetActorLocation()).GetSafeNormal();
234 }
235 }
236
237 DeathImpulse *= launchPower;
238 GetMesh()->AddImpulse(DeathImpulse, NAME_None, true);
239}

254void FAttackTask_ApproachTarget::UpdateTask(float DeltaTime)
255{
256 if (!m_targetActor.IsValid())
257 {
258 SourceBoss->OnPreAttackTaskFinished(TaskType);
259 return;
260 }
261
262 FVector Dir = (m_targetActor->GetActorLocation() - SourceBoss->GetActorLocation());
263 Dir.Z = 0.f;
264 if (!Dir.IsNearlyZero())
265 {
266 SourceBoss->SetActorRotation(Dir.Rotation());
267 }
268
269 if (Dir.Length() <= m_performRange)
270 {
271 SourceBoss->OnPreAttackTaskFinished(TaskType);
272 }
273}

275void FAttackTask_JumpAttack::UpdateTask(float DeltaTime)
276{
277 if (!m_targetActor.IsValid())
278 {
279 SourceBoss->OnAttackTaskConditionMet(TaskType);
280 return;
281 }
282
283 FVector Dir = (m_targetLocation - SourceBoss->GetActorLocation());
284 Dir.Z = 0.0;
285 if (!Dir.IsNearlyZero())
286 {
287 // We keep its Pitch and Roll
288 const FRotator curtRot = SourceBoss->GetActorRotation();
289 const float dirYaw = Dir.Rotation().Yaw;
290 const FRotator newRot = FRotator{curtRot.Pitch, dirYaw, curtRot.Roll};
291 SourceBoss->SetActorRotation(newRot);
292 }
293
294 const float Distance = FVector::Dist2D(SourceBoss->GetActorLocation(), m_targetLocation);
295
296 if (Distance > 50.f)
297 {
298 const FVector destination = SourceBoss->GetActorLocation() + Dir.GetSafeNormal2D() * m_jumpMoveSpeed * DeltaTime;
299 SourceBoss->SetActorLocation(destination);
300 }
301 else
302 {
303 SourceBoss->CurrentSpeed = 0.f;
304 SourceBoss->OnAttackTaskConditionMet(TaskType);
305 }
306}
