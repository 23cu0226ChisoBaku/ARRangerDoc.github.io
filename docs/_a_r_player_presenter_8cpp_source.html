201void UARPlayerPresenter::HandleCharacterMove(double InX, double InY, double InDeadZone, double InMinInput)
202{
203 // 入力値の絶対値をチェックしてデッドゾーン以下は0に
204 const double radiusSquared = FMath::Square(InX) + FMath::Square(InY);
205 const double moveDeadZoneSquared = FMath::Square(FMath::Max(0.0, InDeadZone));
206
207 // デッドゾーン以下
208 if (radiusSquared <= moveDeadZoneSquared)
209 {
210 return;
211 }
212
213 const double realMinInput = FMath::Min(InMinInput, MAX_INPUT_VALUE);
214 // インプット閾値レベル
215 const TArray<double> inputThresholdLevel{
216 realMinInput, // LV1 : 最小入力値
217 MAX_INPUT_VALUE, // LVMax: 最大入力値
218 };
219
220 double adjustedForward = InY;
221 double adjustedRight = InX;
222 for (int32 idx = 0; idx < inputThresholdLevel.Num(); ++idx)
223 {
224 // インプット閾値まで補正する
225 const double inputLevelValue = inputThresholdLevel[idx];
226 const double inputLevelValueSquared = FMath::Square(inputLevelValue);
227 if (radiusSquared < inputLevelValueSquared)
228 {
229 const double inputModifier = inputLevelValue / FMath::Sqrt(radiusSquared);
230 adjustedForward *= inputModifier;
231 adjustedRight *= inputModifier;
232 break;
233 }
234 }
235
236 if (ViewCharacter != nullptr)
237 {
238 // Move character
239 ViewCharacter->DoMove(adjustedRight, adjustedForward);
240 }
241}

264void UARPlayerPresenter::RegisterSnapTargetTask()
265{
266 if (ViewCharacter == nullptr)
267 {
268 return;
269 }
270
271 Model.SnapTargetActor.Reset();
272 Model.TargetSnapInputDirection.Normalize();
273
274 const FVector curtPlayerDir = ViewCharacter->GetActorForwardVector();
275 const FRotator curtCtrlRotation = ViewCharacter->GetController()->GetControlRotation();
276 const FRotator yawRotation{0.0, curtCtrlRotation.Yaw, 0.0};
277 const FVector forwardDirection = FRotationMatrix(yawRotation).GetUnitAxis(EAxis::X);
278 const FVector rightDirection = FRotationMatrix(yawRotation).GetUnitAxis(EAxis::Y);
279 const FVector targetPlayerDir = ((forwardDirection * Model.TargetSnapInputDirection.Y) + (rightDirection * Model.TargetSnapInputDirection.X)).GetSafeNormal();
280 const FVector startLoc = ViewCharacter->GetActorLocation();
281 const FVector endLoc = startLoc + targetPlayerDir * Model.TargetSnapDetectMaxRange;
282
283 // NOTE Magic number is bad
284 float radius = 200.f;
285 if (UCapsuleComponent* capsule = ViewCharacter->GetCapsuleComponent())
286 {
287 radius = capsule->GetScaledCapsuleHalfHeight();
288 }
289
290 // TODO
291 TArray< TEnumAsByte<EObjectTypeQuery> > objTypes{};
292 objTypes.Add(UEngineTypes::ConvertToObjectType(ECC_WorldStatic));
293 objTypes.Add(UEngineTypes::ConvertToObjectType(ECC_WorldDynamic));
294 objTypes.Add(UEngineTypes::ConvertToObjectType(ECC_Pawn));
295 objTypes.Add(UEngineTypes::ConvertToObjectType(ECC_PhysicsBody));
296 TArray<AActor*> ignoreActors{};
297 ignoreActors.Add(ViewCharacter);
298
299 TArray<FHitResult> outResults{};
300
301 // TODO Range Detectorなど既存の機能を使う
302 const bool bHit = UKismetSystemLibrary::SphereTraceMultiForObjects(
303 this,
304 startLoc,
305 endLoc,
306 radius,
307 objTypes,
308 false, // bTraceComplex
309 ignoreActors,
310 EDrawDebugTrace::Persistent,
311 outResults,
312 true // bIgnoreSelf
313 );
314
315 if (bHit)
316 {
317 const FVector playerLoc = startLoc;
318 for (const FHitResult& hitResult : outResults)
319 {
320 AActor* hitActor = hitResult.GetActor();
321 if (hitActor == nullptr)
322 {
323 continue;
324 }
325
326 // Ignore actors that hit by hemisphere behind player's face direction
327 const FVector dirToTarget = (hitActor->GetActorLocation() - ViewCharacter->GetActorLocation()).GetSafeNormal();
328 if (FVector::DotProduct(dirToTarget, targetPlayerDir) < 0.0f)
329 {
330 continue;
331 }
332
333 if (hitActor->GetClass()->ImplementsInterface(UARAttackable::StaticClass()))
334 {
335 IARAttackable* attackable = ::Cast<IARAttackable>(hitActor);
336 if (attackable == nullptr)
337 {
338 continue;
339 }
340
341 const float curtHitResultDistanceSquared = (playerLoc - hitResult.GetActor()->GetActorLocation()).SquaredLength();
342 if (Model.SnapTargetActor.IsValid())
343 {
344 // Find min distance to player
345 const float curtMinDistanceSquared = (playerLoc - Model.SnapTargetActor->GetActorLocation()).SquaredLength();
346 if (curtMinDistanceSquared > curtHitResultDistanceSquared)
347 {
348 Model.SnapTargetActor = hitResult.GetActor();
349 UPrimitiveComponent* targetPrimitiveComp = hitResult.GetComponent();
350 const FVector relativeImpactPoint = hitResult.ImpactPoint - targetPrimitiveComp->GetComponentLocation();
351 // We do not use Z-component of impact point
352 Model.SnapTargetImpactPoint = FVector{relativeImpactPoint.X, relativeImpactPoint.Y, 0.0};
353 }
354 }
355 else
356 {
357 Model.SnapTargetActor = hitResult.GetActor();
358 UPrimitiveComponent* targetPrimitiveComp = hitResult.GetComponent();
359 const FVector relativeImpactPoint = hitResult.ImpactPoint - targetPrimitiveComp->GetComponentLocation();
360 // We do not use Z-component of impact point
361 Model.SnapTargetImpactPoint = FVector{relativeImpactPoint.X, relativeImpactPoint.Y, 0.0};
362 }
363
364 Model.bCanUpdateSnapMovement = true;
365 }
366 }
367
368 if (Model.bCanUpdateSnapMovement)
369 {
370 Model.SnapStartPosition = ViewCharacter->GetActorLocation();
371 Model.SnapStartRotation = ViewCharacter->GetActorRotation();
372 Model.bIsReadyToSearchSnapTarget = false;
373
374 Handle_UpdateSnapTarget = ViewCharacter->TickTaskDelegate.AddUObject(this, &ThisClass::UpdateSnapTarget);
375 }
376 }

379void UARPlayerPresenter::UpdateSnapTarget(float DeltaTime)
380{
381 if (ViewCharacter == nullptr)
382 {
383 return;
384 }
385
386 FVector newLocation = ViewCharacter->GetActorLocation();
387 FRotator newRotation = ViewCharacter->GetActorRotation();
388
389 if ((Model.SnapTargetActor.IsValid()) &&
390 !FMath::IsNearlyZero(Model.SnapTimeInterval) &&
391 (Model.SnapTimeCounter < Model.SnapTimeInterval)
392 )
393 {
394 const float alphaMin = 0.0f;
395 const float alphaMax = 1.0f;
396
397 Model.SnapTimeCounter += DeltaTime;
398
399 // Calculate new rotation
400 const float RotateInterpInterval = Model.SnapTimeInterval / 2.0f;
401 FVector faceToTarget_XYAxis = (Model.SnapTargetActor->GetActorLocation() - ViewCharacter->GetActorLocation());
402 faceToTarget_XYAxis.Z = 0.0;
403 const FRotator faceToTargetRot = faceToTarget_XYAxis.Rotation();
404 const float rotLerpAlpha = FMath::Clamp((Model.SnapTimeCounter / RotateInterpInterval), alphaMin, alphaMax);
405 newRotation = FMath::InterpCircularOut(Model.SnapStartRotation, faceToTargetRot, rotLerpAlpha);
406
407 // Calculate new location
408 const FVector playerToTargetOffset = Model.SnapTargetImpactPoint.GetSafeNormal() * ViewCharacter->GetCapsuleComponent()->GetScaledCapsuleRadius();
409 const FVector newTargetLocation = Model.SnapTargetActor->GetActorLocation() + Model.SnapTargetImpactPoint + playerToTargetOffset;
410 const FVector newTargetLocation_UsePlayerZ = FVector{newTargetLocation.X, newTargetLocation.Y, ViewCharacter->GetActorLocation().Z};
411 const float locLerpAlpha = FMath::Clamp((Model.SnapTimeCounter / Model.SnapTimeInterval), alphaMin, alphaMax);
412 newLocation = FMath::InterpCircularIn(Model.SnapStartPosition, newTargetLocation_UsePlayerZ, locLerpAlpha);
413 }
414 // Stop snapping
415 else
416 {
417 StopSnapTargetInternal();
418 }
419
420 ViewCharacter->OnTargetSnapped(newLocation, newRotation);
421
422}

511void UARPlayerPresenter::HandleTransformedEvent(EARMagnetismType InNewTransformation)
512{
513 if (ViewCharacter == nullptr)
514 {
515 return;
516 }
517
518 switch (InNewTransformation)
519 {
520 case EARMagnetismType::Attraction:
521 {
522 // Try to climb
523 TArray<UPrimitiveComponent*> OverlappingComps{};
524 ViewCharacter->GetOverlappingComponents(OverlappingComps);
525
526 for (UPrimitiveComponent* Comp : OverlappingComps)
527 {
528 if (Comp->GetOwner()->IsA<AInsekiClimbingObject>())
529 {
530 StartClimbing();
531 break;
532 }
533 }
534 }
535 break;
536
537 // Try triggering physics event
538 case EARMagnetismType::Repulsion:
539 {
540 const UWorld* world = GetWorld();
541 UCapsuleComponent* capsuleComp = ViewCharacter->GetCapsuleComponent();
542
543 // Use player character CapsuleComponent as collision shape.Make it a bit larger than origin shape
544 // TODO Maybe we can remove this magic number?
545 const float extendCapsuleRadius = 5.f;
546 const float extendCapsuleHalfHeight = 5.f;
547 const float shapeRadius = capsuleComp->GetScaledCapsuleRadius() + extendCapsuleRadius;
548 const float shapeHalfHeight = capsuleComp->GetScaledCapsuleHalfHeight() + extendCapsuleHalfHeight;
549 const FCollisionShape characterCapsuleShape = FCollisionShape::MakeCapsule(shapeRadius, shapeHalfHeight);
550 const FVector origin = capsuleComp->GetComponentLocation();
551 const FQuat originQuat = capsuleComp->GetComponentQuat();
552
553 FCollisionObjectQueryParams objQueryParams{};
554 objQueryParams.AddObjectTypesToQuery(ECC_WorldStatic);
555 objQueryParams.AddObjectTypesToQuery(ECC_WorldDynamic);
556 objQueryParams.AddObjectTypesToQuery(ECC_Pawn);
557 objQueryParams.AddObjectTypesToQuery(ECC_PhysicsBody);
558
559 FCollisionQueryParams queryParams{SCENE_QUERY_STAT(HandleTransformedEvent), false, ViewCharacter};
560 queryParams.bReturnPhysicalMaterial = true;
561
562 TArray<FHitResult> outHitResult{};
563 const bool bHit = world->SweepMultiByObjectType(
564 outHitResult,
565 origin,
566 origin,
567 originQuat,
568 objQueryParams,
569 characterCapsuleShape,
570 queryParams);
571
572 if (bHit)
573 {
574 for (const FHitResult& hit : outHitResult)
575 {
576 AActor* hitActor = hit.GetActor();
577 if (IARMagnetizableInterface* magnetizableObj = Cast<IARMagnetizableInterface>(hitActor))
578 {
579 Physics_RegisterMagneticTask_Once(ViewCharacter, magnetizableObj);
580 }
581 }
582 }
583 }
584 break;
585 }
586}

604void UARPlayerPresenter::StartClimbing()
605{
606 const bool bCanStartClimb = !Model.bIsClimbing
607 && (ViewCharacter != nullptr)
608 && (ViewCharacter->GetMagnetismType() == EARMagnetismType::Attraction);
609 if (!bCanStartClimb)
610 {
611 return;
612 }
613
614 Model.bIsClimbing = true;
615 ViewCharacter->OnClimbStarted();
616 Handle_UpdateClimbing = ViewCharacter->TickTaskDelegate.AddUObject(this, &ThisClass::UpdateClimbing);
617
618 // 壁があるかを判定
619 const FVector Start = ViewCharacter->GetActorLocation();
620 // FIXME Magic number is bad
621 const FVector End = Start + ViewCharacter->GetActorForwardVector() * 100.0f;
622
623 FHitResult HitResult{};
624 FCollisionQueryParams Params{};
625 Params.AddIgnoredActor(ViewCharacter);
626 const bool bHit = GetWorld()->LineTraceSingleByChannel(HitResult, Start, End, ECC_Visibility, Params);
627 if (!bHit)
628 {
629 StopClimbing();
630 return;
631 }
632
633 // 壁の法線を保存
634 Model.ClimbSurfaceNormal = HitResult.ImpactNormal;
635}

757bool UARPlayerPresenter::CanUpdateClimbingInternal() const
758{
759 const bool bCanUpdateClimbing = Model.bIsClimbing
760 && (ViewCharacter != nullptr);
761 if (!bCanUpdateClimbing)
762 {
763 return false;
764 }
765
766 // 壁回転処理
767 // 足元の位置（Capsuleの底の位置）
768 const UCapsuleComponent* capsuleComp = ViewCharacter->GetCapsuleComponent();
769 const float capsuleHalfHeight = capsuleComp->GetScaledCapsuleHalfHeight();
770 const FVector curtLocation = ViewCharacter->GetActorLocation();
771 // 壁に対して垂直な向きに少しめり込むようにして設定
772 // FIXME Remove magic number
773 const FVector Start = curtLocation - Model.ClimbSurfaceNormal * (capsuleHalfHeight - 5.0f);
774 const FVector End = Start - Model.ClimbSurfaceNormal * 7.0f;
775
776 FHitResult HitResult{};
777 FCollisionQueryParams Params{};
778 Params.AddIgnoredActor(ViewCharacter);
779
780 const bool bHit = GetWorld()->LineTraceSingleByChannel(HitResult, Start, End, ECC_Visibility, Params);
781 const bool bCanKeepClimbingState = bHit && ViewCharacter->GetMagnetismType() == EARMagnetismType::Attraction;
782
783 return bCanKeepClimbingState;
784}
