69 ViewCharacter->AttackAbilityEndDelegate.AddUObject(this, &ThisClass::HandleAttackAbilityEnded);

195void UARPlayerPresenter::HandleCharacterMove(double InX, double InY, double InDeadZone, double InMinInput)
196{
197 // 入力値の絶対値をチェックしてデッドゾーン以下は0に
198 const double radiusSquared = FMath::Square(InX) + FMath::Square(InY);
199 const double moveDeadZoneSquared = FMath::Square(FMath::Max(0.0, InDeadZone));
200
201 // デッドゾーン以下の入力を受け付けない
202 if ((radiusSquared <= moveDeadZoneSquared) || FMath::IsNearlyZero(radiusSquared))
203 {
204 return;
205 }
206
207 // Input max value(scalar)
208 constexpr double MAX_INPUT_VALUE = 1.0;
209
210 const double realMinInput = FMath::Min(InMinInput, MAX_INPUT_VALUE);
211 // インプット閾値レベル
212 const TArray<double> inputThresholdLevel{
213 realMinInput, // LV1 : 最小入力値
214 MAX_INPUT_VALUE, // LVMax: 最大入力値
215 };
216
217 double adjustedForward = InY;
218 double adjustedRight = InX;
219 for (int32 idx = 0; idx < inputThresholdLevel.Num(); ++idx)
220 {
221 // インプット閾値まで補正する
222 const double inputLevelValue = inputThresholdLevel[idx];
223 const double inputLevelValueSquared = FMath::Square(inputLevelValue);
224 if (radiusSquared < inputLevelValueSquared)
225 {
226 const double inputModifier = inputLevelValue / FMath::Sqrt(radiusSquared);
227 adjustedForward *= inputModifier;
228 adjustedRight *= inputModifier;
229 break;
230 }
231 }
232
233 if (ViewCharacter != nullptr)
234 {
235 ViewCharacter->DoMove(adjustedRight, adjustedForward);
236 }
237}

260void UARPlayerPresenter::RegisterSnapTargetTask()
261{
262 if (ViewCharacter == nullptr)
263 {
264 return;
265 }
266
267 Model.SnapTargetActor.Reset();
268 Model.TargetSnapInputDirection.Normalize();
269
270 const FVector curtPlayerDir = ViewCharacter->GetActorForwardVector();
271 const FRotator curtCtrlRotation = ViewCharacter->GetController()->GetControlRotation();
272 const FRotator yawRotation{0.0, curtCtrlRotation.Yaw, 0.0};
273 const FVector forwardDirection = FRotationMatrix(yawRotation).GetUnitAxis(EAxis::X);
274 const FVector rightDirection = FRotationMatrix(yawRotation).GetUnitAxis(EAxis::Y);
275 const FVector targetPlayerDir = ((forwardDirection * Model.TargetSnapInputDirection.Y) + (rightDirection * Model.TargetSnapInputDirection.X)).GetSafeNormal();
276 const FVector startLoc = ViewCharacter->GetActorLocation();
277 const FVector endLoc = startLoc + targetPlayerDir * Model.TargetSnapDetectMaxRange;
278 const float radius = ViewCharacter->GetCapsuleComponent()->GetScaledCapsuleHalfHeight();
279
280 // TODO Range Detectorなど既存の機能を使う
281 TArray< TEnumAsByte<EObjectTypeQuery> > objTypes{};
282 objTypes.Add(UEngineTypes::ConvertToObjectType(ECC_WorldStatic));
283 objTypes.Add(UEngineTypes::ConvertToObjectType(ECC_WorldDynamic));
284 objTypes.Add(UEngineTypes::ConvertToObjectType(ECC_Pawn));
285 objTypes.Add(UEngineTypes::ConvertToObjectType(ECC_PhysicsBody));
286 TArray<AActor*> ignoreActors{};
287 ignoreActors.Add(ViewCharacter);
288 TArray<FHitResult> outResults{};
289
290 const bool bHit = UKismetSystemLibrary::SphereTraceMultiForObjects(
291 ViewCharacter,
292 startLoc,
293 endLoc,
294 radius,
295 objTypes,
296 false, // bTraceComplex
297 ignoreActors,
298 EDrawDebugTrace::Persistent,
299 outResults,
300 true // bIgnoreSelf
301 );
302
303 if (bHit)
304 {
305 const FVector playerLoc = startLoc;
306 for (const FHitResult& hitResult : outResults)
307 {
308 AActor* hitActor = hitResult.GetActor();
309 if (hitActor == nullptr)
310 {
311 continue;
312 }
313
314 // プレイヤーを始点とした検索範囲のうち、プレイヤーの後ろの半円に入っているターゲットを無視する
315 const FVector dirToTarget = (hitActor->GetActorLocation() - ViewCharacter->GetActorLocation()).GetSafeNormal();
316 if (FVector::DotProduct(dirToTarget, targetPlayerDir) < 0.0f)
317 {
318 continue;
319 }
320
321 // 攻撃できないターゲットに吸着しない
322 if (IARAttackable::IsActorAttackable(hitActor))
323 {
324 const float curtHitResultDistanceSquared = (playerLoc - hitResult.GetActor()->GetActorLocation()).SquaredLength();
325 if (Model.SnapTargetActor.IsValid())
326 {
327 const float curtMinDistanceSquared = (playerLoc - Model.SnapTargetActor->GetActorLocation()).SquaredLength();
328 // 前の吸着対象より近い対象に更新
329 if (curtMinDistanceSquared > curtHitResultDistanceSquared)
330 {
331 UpdateSnapTargetImpactPoint(hitResult.GetComponent(), hitResult.ImpactPoint);
332 }
333 }
334 else
335 {
336 UpdateSnapTargetImpactPoint(hitResult.GetComponent(), hitResult.ImpactPoint);
337 }
338
339 Model.bCanUpdateSnapMovement = true;
340 }
341 }
342
343 if (Model.bCanUpdateSnapMovement)
344 {
345 Model.SnapStartPosition = ViewCharacter->GetActorLocation();
346 Model.SnapStartRotation = ViewCharacter->GetActorRotation();
347 Model.bIsReadyToSearchSnapTarget = false;
348
349 Handle_UpdateSnapTarget = ViewCharacter->TickTaskDelegate.AddUObject(this, &ThisClass::UpdateSnapTarget);
350 }
351 }
352}

354void UARPlayerPresenter::UpdateSnapTarget(float DeltaTime)
355{
356 if (ViewCharacter == nullptr)
357 {
358 return;
359 }
360
361 FVector newLocation = ViewCharacter->GetActorLocation();
362 FRotator newRotation = ViewCharacter->GetActorRotation();
363
364 if ((Model.SnapTargetActor.IsValid()) &&
365 !FMath::IsNearlyZero(Model.SnapTimeInterval) &&
366 (Model.SnapTimeCounter < Model.SnapTimeInterval)
367 )
368 {
369 const float alphaMin = 0.0f;
370 const float alphaMax = 1.0f;
371
372 Model.SnapTimeCounter += DeltaTime;
373
374 // 吸着先に向ける回転を計算する
375 {
376 const float RotateInterpInterval = Model.SnapTimeInterval / 2.0f;
377 FVector faceToTarget_XYAxis = (Model.SnapTargetActor->GetActorLocation() - ViewCharacter->GetActorLocation());
378 faceToTarget_XYAxis.Z = 0.0;
379 const FRotator faceToTargetRot = faceToTarget_XYAxis.Rotation();
380 const float rotLerpAlpha = FMath::Clamp((Model.SnapTimeCounter / RotateInterpInterval), alphaMin, alphaMax);
381 newRotation = FMath::InterpCircularOut(Model.SnapStartRotation, faceToTargetRot, rotLerpAlpha);
382 }
383
384 // 吸着先に向ける座標を計算する
385 {
386 const FVector playerToTargetOffset = Model.SnapTargetImpactPoint.GetSafeNormal() * ViewCharacter->GetCapsuleComponent()->GetScaledCapsuleRadius();
387 const FVector newTargetLocation = Model.SnapTargetActor->GetActorLocation() + Model.SnapTargetImpactPoint + playerToTargetOffset;
388 const FVector newTargetLocation_UsePlayerZ = FVector{newTargetLocation.X, newTargetLocation.Y, ViewCharacter->GetActorLocation().Z};
389 const float locLerpAlpha = FMath::Clamp((Model.SnapTimeCounter / Model.SnapTimeInterval), alphaMin, alphaMax);
390 newLocation = FMath::InterpCircularIn(Model.SnapStartPosition, newTargetLocation_UsePlayerZ, locLerpAlpha);
391 }
392 }
393 // 吸着処理を止める
394 else
395 {
396 StopSnapTargetInternal();
397 }
398
399 ViewCharacter->OnTargetSnapped(newLocation, newRotation);
400}

415void UARPlayerPresenter::HandleCharacterChargeRotate(double InX, double InY)
416{
417 // 使わない
418 (void)InY;
419
420 const bool bCanRotate = (ViewCharacter != nullptr) && !Model.bIsInAir && Model.bIsCharging;
421 if (!bCanRotate)
422 {
423 return;
424 }
425
426 // 回転範囲を絞る
427 const FRotator curtPlayerDir_Rot = ViewCharacter->GetActorRotation();
428 const double curtYawOffsetToChargeStart = curtPlayerDir_Rot.Yaw - Model.ChargeStartFaceDir.Rotation().Yaw;
429 double rotate_Yaw = InX;
430 if (FMath::Abs(curtYawOffsetToChargeStart + rotate_Yaw) > Model.ChargeRotateHalfRange)
431 {
432 rotate_Yaw = FMath::Sign(rotate_Yaw) * (Model.ChargeRotateHalfRange - FMath::Abs(curtYawOffsetToChargeStart));
433 }
434
435 ViewCharacter->DoRotate(rotate_Yaw);
436}

438void UARPlayerPresenter::HandleBattleResult(AARRangerCharacter* InAffectedCharacter, const ARRanger::Battle::FARDamageResult& InDamageResult)
439{
440 check(InAffectedCharacter == ViewCharacter);
441
442 // ダメージが正の数なので、HPの変化量を負の数にする
443 const float HPChangeValue = -InDamageResult.FinalDamage;
444 bool bIsDead = false;
445 if (Model.HealthComponent != nullptr)
446 {
447 Model.HealthComponent->HandleHealthChange(InDamageResult.Instigator, HPChangeValue);
448
449 // ノックバック処理
450 if (Model.HealthComponent->GetHealth() > 0.0f)
451 {
452 constexpr double launchPowerFactor = 1.0;
453
454 FVector launchDirNorm = InDamageResult.FinalLaunchDirection;
455 launchDirNorm.Z = 0.0;
456 launchDirNorm.Normalize();
457 if (ViewCharacter != nullptr)
458 {
459 const double finalLaunchPower = Model.LaunchPower * launchPowerFactor;
460 ViewCharacter->LaunchCharacter_Ext(launchDirNorm, finalLaunchPower);
461 }
462 }
463 else
464 {
465 bIsDead = true;
466 }
467 }
468
469 if (ViewCharacter != nullptr)
470 {
471 ViewCharacter->OnHealthChanged(InDamageResult.Instigator, HPChangeValue, bIsDead);
472 }

485void UARPlayerPresenter::HandleTransformedEvent(EARMagnetismType InNewTransformation)
486{
487 if (ViewCharacter == nullptr)
488 {
489 return;
490 }
491
492 // 変身した瞬間、一回だけ周りに磁力オブジェクトあるかを確認する、あったらタスクを登録する
493 switch (InNewTransformation)
494 {
495 // 引力状態に変身
496 case EARMagnetismType::Attraction:
497 {
498 // Try to climb
499 TArray<UPrimitiveComponent*> OverlappingComps{};
500 ViewCharacter->GetOverlappingComponents(OverlappingComps);
501
502 for (UPrimitiveComponent* Comp : OverlappingComps)
503 {
504 if (Comp->GetOwner()->IsA<AInsekiClimbingObject>())
505 {
506 StartClimbing();
507 break;
508 }
509 }
510 }
511 break;
512
513 // 斥力状態に変身
514 case EARMagnetismType::Repulsion:
515 {
516 const UWorld* world = GetWorld();
517 UCapsuleComponent* capsuleComp = ViewCharacter->GetCapsuleComponent();
518
519 // Use player character CapsuleComponent as collision shape.Make it a bit larger than origin shape
520 const float extendCapsuleRadius = 5.f;
521 const float extendCapsuleHalfHeight = 5.f;
522 const float shapeRadius = capsuleComp->GetScaledCapsuleRadius() + extendCapsuleRadius;
523 const float shapeHalfHeight = capsuleComp->GetScaledCapsuleHalfHeight() + extendCapsuleHalfHeight;
524 const FCollisionShape characterCapsuleShape = FCollisionShape::MakeCapsule(shapeRadius, shapeHalfHeight);
525 const FVector origin = capsuleComp->GetComponentLocation();
526 const FQuat originQuat = capsuleComp->GetComponentQuat();
527
528 // TODO Use RangeDetector
529 FCollisionObjectQueryParams objQueryParams{};
530 objQueryParams.AddObjectTypesToQuery(ECC_WorldStatic);
531 objQueryParams.AddObjectTypesToQuery(ECC_WorldDynamic);
532 objQueryParams.AddObjectTypesToQuery(ECC_Pawn);
533 objQueryParams.AddObjectTypesToQuery(ECC_PhysicsBody);
534
535 FCollisionQueryParams queryParams{SCENE_QUERY_STAT(HandleTransformedEvent), false, ViewCharacter};
536 queryParams.bReturnPhysicalMaterial = true;
537
538 TArray<FHitResult> outHitResult{};
539 const bool bHit = world->SweepMultiByObjectType(
540 outHitResult,
541 origin,
542 origin,
543 originQuat,
544 objQueryParams,
545 characterCapsuleShape,
546 queryParams);
547
548 if (bHit)
549 {
550 for (const FHitResult& hit : outHitResult)
551 {
552 AActor* hitActor = hit.GetActor();
553 if (IARMagnetizableInterface* magnetizableObj = Cast<IARMagnetizableInterface>(hitActor))
554 {
555 Physics_RegisterMagneticTask_Once(ViewCharacter, magnetizableObj);
556 }
557 }
558 }
559 }
560 break;
561 }
562}

579void UARPlayerPresenter::StartClimbing()
580{
581 const bool bCanStartClimb = !Model.bIsClimbing
582 && (ViewCharacter != nullptr)
583 && (ViewCharacter->GetMagnetismType() == EARMagnetismType::Attraction);
584 if (!bCanStartClimb)
585 {
586 return;
587 }
588
589 Model.bIsClimbing = true;
590 ViewCharacter->OnClimbStarted();
591 Handle_UpdateClimbing = ViewCharacter->TickTaskDelegate.AddUObject(this, &ThisClass::UpdateClimbing);
592
593 // 壁があるかを判定
594 constexpr float wallCheckRayLength = 100.0f;
595 const FVector Start = ViewCharacter->GetActorLocation();
596 const FVector End = Start + ViewCharacter->GetActorForwardVector() * wallCheckRayLength;
597
598 FHitResult HitResult{};
599 FCollisionQueryParams Params{};
600 Params.AddIgnoredActor(ViewCharacter);
601 const bool bHit = GetWorld()->LineTraceSingleByChannel(HitResult, Start, End, ECC_Visibility, Params);
602 if (!bHit)
603 {
604 StopClimbing();
605 return;
606 }
607
608 // 壁の法線を保存
609 Model.ClimbSurfaceNormal = HitResult.ImpactNormal;
610}

742{
743 const bool bCanUpdateClimbing = Model.bIsClimbing
744 && (ViewCharacter != nullptr);
745 if (!bCanUpdateClimbing)
746 {
747 return false;
748 }
749
750 // 壁回転処理
751 // 足元の位置（Capsuleの底の位置）
752 const UCapsuleComponent* capsuleComp = ViewCharacter->GetCapsuleComponent();
753 const float capsuleHalfHeight = capsuleComp->GetScaledCapsuleHalfHeight();
754 const FVector curtLocation = ViewCharacter->GetActorLocation();
755 // 壁に対して垂直な向きに少しめり込むようにして設定
756 // FIXME Remove magic number
757 const FVector Start = curtLocation - Model.ClimbSurfaceNormal * (capsuleHalfHeight - 5.0f);
758 const FVector End = Start - Model.ClimbSurfaceNormal * 7.0f;
759
760 FHitResult HitResult{};
761 FCollisionQueryParams Params{};
762 Params.AddIgnoredActor(ViewCharacter);

764 const bool bHit = GetWorld()->LineTraceSingleByChannel(HitResult, Start, End, ECC_Visibility, Params);
