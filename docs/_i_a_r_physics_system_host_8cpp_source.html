70void IARPhysicsSystemHost::Physics_RequestMagneticTaskImpl(IARMagnetizableInterface* InSource, IARMagnetizableInterface* InTarget, EMagneticTaskFrequency Frequency)
71{
72 if ((InSource == nullptr) || (InTarget == nullptr))
73 {
74 AR_LOG(LogARPhysics, Error, TEXT("Input is Invalid. Caller:[%s]"), *GetNameSafe(InSource->GetActor()));
75 return;
76 }
77
78 FARPhysicsRegistry request;
79 request.Source = InSource;
80 request.Target = InTarget;
81
82 using enum EARMagnetismType;
83 if ((InSource->GetMagnetismType() == Attraction) && (InTarget->GetMagnetismType() == Attraction))
84 {
85 request.Type = EPhysicsRegistryType::RequestAttraction;
86 }
87 else if ((InSource->GetMagnetismType() == Repulsion) && (InTarget->GetMagnetismType() == Repulsion))
88 {
89 request.Type = EPhysicsRegistryType::RequestRepulsion;
90 }
91 else
92 {
93 AR_LOG(LogARPhysics, Warning, TEXT("Request type is NONE."));
94 request.Type = EPhysicsRegistryType::None;
95 return;
96 }
97
98 // TODO 二種類のEnumを利用する手間を減らしたい
99 switch (Frequency)
100 {
101 case Once:
102 {
103 request.Frequency = EPhysicsExecuteFrequency::Once;
104 }
105 break;
106 case Constantly:
107 {
108 request.Frequency = EPhysicsExecuteFrequency::Constantly;
109 }
110 break;
111 }
112
113 GetEngine().RegisterPhysicsTask(request);
114}
