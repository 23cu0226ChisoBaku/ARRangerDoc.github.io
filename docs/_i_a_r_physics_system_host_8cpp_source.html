62void IARPhysicsSystemHost::Physics_RequestMagneticTaskImpl(IARMagnetizableInterface* InSource, IARMagnetizableInterface* InTarget, EPhysicsExecuteFrequency Frequency)
63{
64 if ((InSource == nullptr) || (InTarget == nullptr))
65 {
66 AR_LOG(LogARPhysics, Error, TEXT("Input is Invalid. Caller:[%s]"), *GetNameSafe(InSource->GetActor()));
67 return;
68 }
69
70 FARPhysicsRegistry request{};
71 request.Source = InSource;
72 request.Target = InTarget;
73
74 using enum EARMagnetismType;
75 // 同じタイプの磁力オブジェクトかつNoneじゃないタイプだとリクエストを拒否する
76 if (InSource->GetMagnetismType() != InTarget->GetMagnetismType() || (InSource->GetMagnetismType() == None))
77 {
78 AR_LOG(LogARPhysics, Warning, TEXT("MagnetismType is not same or one of the type is None.Request denied."));
79 return;
80 }
81
82 // 磁力リクエストタイプ設定
83 if (InSource->GetMagnetismType() == Attraction)
84 {
85 request.Type = EPhysicsRegistryType::RequestAttraction;
86 }
87 else if (InSource->GetMagnetismType() == Repulsion)
88 {
89 request.Type = EPhysicsRegistryType::RequestRepulsion;
90 }
91
92 request.Frequency = Frequency;
93
94 GetPhysicsSystem().RegisterPhysicsTask(request);
95}
