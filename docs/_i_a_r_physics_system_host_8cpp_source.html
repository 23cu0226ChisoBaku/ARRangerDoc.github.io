63void IARPhysicsSystemHost::Physics_RequestMagneticTaskImpl(IARMagnetizableInterface* InSource, IARMagnetizableInterface* InTarget, EPhysicsExecuteFrequency Frequency)
64{
65 if ((InSource == nullptr) || (InTarget == nullptr))
66 {
67 AR_LOG(LogARPhysics, Error, TEXT("Input is Invalid. Caller:[%s]"), *GetNameSafe(InSource->GetActor()));
68 return;
69 }
70
71 FARPhysicsRegistry request{};
72 request.Source = InSource;
73 request.Target = InTarget;
74
75 using enum EARMagnetismType;
76 if ((InSource->GetMagnetismType() == Attraction) && (InTarget->GetMagnetismType() == Attraction))
77 {
78 request.Type = EPhysicsRegistryType::RequestAttraction;
79 }
80 else if ((InSource->GetMagnetismType() == Repulsion) && (InTarget->GetMagnetismType() == Repulsion))
81 {
82 request.Type = EPhysicsRegistryType::RequestRepulsion;
83 }
84 else
85 {
86 AR_LOG(LogARPhysics, Warning, TEXT("Request type is NONE."));
87 request.Type = EPhysicsRegistryType::None;
88 return;
89 }
90
91 request.Frequency = Frequency;
92
93 GetEngine().RegisterPhysicsTask(request);
94}
