165void AARGuidancePoint::UpdateAnim(const FVector& Destination, float DeltaTime)
166{
167 m_guidanceAnimTimeCnt += DeltaTime;
168 m_guidanceAnimTimeCnt = FMath::Clamp(m_guidanceAnimTimeCnt, 0.0f, GuidanceAnimLength);
169
170 // Update Animation On BP
171 K2_OnGuidanceAnimationUpdated(GetActorLocation(), Destination, GuidanceAnimLength, m_guidanceAnimTimeCnt, DeltaTime);
172
173 if (FMath::IsNearlyEqual(m_guidanceAnimTimeCnt, GuidanceAnimLength))
174 {
175 // Stop Animation
176 StopAnim();
177
178 if (!FMath::IsNearlyZero(GuidanceAnimDelay))
179 {
180 bIsAnimDelaying = true;
181
182 auto delayHandler = [this]()
183 {
184 this->ResetAnim();
185 };
186
187 if (!m_guidanceAnimDelayTimerHandle.IsValid())
188 {
189 GetWorld()->GetTimerManager().SetTimer(m_guidanceAnimDelayTimerHandle, delayHandler, GuidanceAnimDelay, false);
190 }
191 }
192 }
193}
