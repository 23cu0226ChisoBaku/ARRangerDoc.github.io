159void AARPhysicsTickProcessorActor::UnregisterQueuedTickObject()
160{
161 for (const auto& queuedTickObj : UnregisterTickObjectQueue)
162 {
163 if (queuedTickObj != nullptr)
164 {
165 queuedTickObj->UnregisterPhysicsTickFunction();
166 auto searchFunctor = [&queuedTickObj](const FARMagneticTickObjectEntry& Entry)
167 {
168 return Entry.TickObject == queuedTickObj;
169 };
170
171 FARMagneticTickObjectEntry* foundEntry = MagneticTickObjectEntries.FindByPredicate(searchFunctor);
172 if (foundEntry != nullptr)
173 {
174 foundEntry->TickObject->ConditionalBeginDestroy();
175 MagneticTickObjectEntries.RemoveSingle(*foundEntry);
176 }
177 }
178 }
179
180 UnregisterTickObjectQueue.Reset();
181}

183void AARPhysicsTickProcessorActor::RegisterMagneticTarget(IARMagnetizableInterface* InTarget, IARMagnetizableInterface* InAffectedObj, EPhysicsRegistryType InRequestType, EPhysicsExecuteFrequency InFrequency)
184{
185 if (InTarget == nullptr)
186 {
187 AR_LOG(LogARPhysics, Warning, TEXT("Magnetizable target is invalid. Register denied"))
188 return;
189 }
190
191 FARMagneticTickObjectEntry* foundEntry = GetMagneticTickObjectEntry(InTarget);
192 if (foundEntry == nullptr)
193 {
194 TSubclassOf<UARMagneticTickObject> allocateClass = nullptr;
195 // TODO
196 using enum EPhysicsRegistryType;
197 switch (InRequestType)
198 {
199 case RequestAttraction:
200 {
201 allocateClass = AttractionTickClass;
202 }
203 break;
204 case RequestRepulsion:
205 {
206 allocateClass = RepulsionTickClass;
207 }
208 break;
209 }
210
211 foundEntry = AllocateMagneticTickObject(InTarget, allocateClass);
212 }
213
214 check(foundEntry != nullptr);
215 foundEntry->AffectedObjectInterfaces.AddUnique(InAffectedObj);
216
217 switch (InFrequency)
218 {
219 case EPhysicsExecuteFrequency::Once:
220 {
221 foundEntry->TickObject->SetFrequency(EARPhysicsTickFrequency::TF_Once);
222 }
223 break;
224
225 case EPhysicsExecuteFrequency::Constantly:
226 {
227 foundEntry->TickObject->SetFrequency(EARPhysicsTickFrequency::TF_Default);
228 }
229 break;
230
231 // Not allowed Frequency that is not implemented
232 default:
233 {
234 check(false);
235 }
236 break;
237 }
238
239
240}

242void AARPhysicsTickProcessorActor::UnregisterMagneticTarget(IARMagnetizableInterface* InSource, IARMagnetizableInterface* InTarget)
243{
244 check(InSource != nullptr);
245 check(InTarget != nullptr);
246
247 FARMagneticTickObjectEntry* foundEntry = GetMagneticTickObjectEntry(InSource);
248 if (foundEntry != nullptr)
249 {
250 foundEntry->AffectedObjectInterfaces.RemoveSingle(InTarget);
251 if (foundEntry->AffectedObjectInterfaces.Num() < 1)
252 {
253 UnregisterTickObjectQueue.Emplace(foundEntry->TickObject);
254 }
255 }

258FARMagneticTickObjectEntry* AARPhysicsTickProcessorActor::GetMagneticTickObjectEntry(IARMagnetizableInterface* InTarget)
259{
260 check(InTarget != nullptr);
261 auto searchFunctor = [InTarget](const FARMagneticTickObjectEntry& Entry)
262 {
263 return Entry.TickObject->IsSameTarget(InTarget);
264 };
265
266 return MagneticTickObjectEntries.FindByPredicate(searchFunctor);
267}

269FARMagneticTickObjectEntry* AARPhysicsTickProcessorActor::AllocateMagneticTickObject(IARMagnetizableInterface* Target, TSubclassOf<UARMagneticTickObject> MagneticTickObjectClass)
270{
271 check(MagneticTickObjectClass != nullptr);
272 UARMagneticTickObject* newTickObject = NewObject<UARMagneticTickObject>(this, MagneticTickObjectClass);
273 check(newTickObject != nullptr);
274
275 newTickObject->SetTargetObject(Target);
276 newTickObject->SetPhysicsEngineProxy(OwningPhysicsEngine->GetProxy());
277
278 RegisterTickObjectQueue.Emplace(newTickObject);
279
280 FARMagneticTickObjectEntry newEntry;
281 newEntry.TickObject = newTickObject;
282 int32 index = MagneticTickObjectEntries.Emplace(newEntry);
283
284 // TODO Mayby dangerous
285 return &MagneticTickObjectEntries[index];
286}
