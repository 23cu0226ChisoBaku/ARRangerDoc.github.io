57void UARPawnInitComponent::InitializeAbilitySystem(UARAbilitySystemComponent* InASC, AActor* InOwnerActor)
58{
59 check(InASC != nullptr);
60 check(InOwnerActor != nullptr);
61
62 if (AbilitySystemComponent == InASC)
63 {
64 return;
65 }
66
67 // Clean up previous ASC
68 if (AbilitySystemComponent != nullptr)
69 {
70 UninitializeAbilitySystem();
71 }
72
73 APawn* ownerPawn = GetPawnChecked<APawn>();
74 AActor* existingAvatar = InASC->GetAvatarActor();
75
76 // Clean up previous avatar ASC if it is acting as the ASC's avatar
77 if ((existingAvatar != nullptr) && (existingAvatar != ownerPawn))
78 {
79 if (UARPawnInitComponent* avatarPawnInitComp = FindPawnInitComponent(existingAvatar))
80 {
81 avatarPawnInitComp->UninitializeAbilitySystem();
82 }
83 }
84
85 AbilitySystemComponent = InASC;
86 AbilitySystemComponent->InitAbilityActorInfo(InOwnerActor, ownerPawn);
87
88 if (PawnInitData != nullptr)
89 {
90 // TODO Make inputID and abilityLevel hard coding temporary
91 int32 inputID = 0;
92 const int32 abilityLevel = 1;
93 for (TSoftClassPtr<UARGameplayAbilityBase> GA : PawnInitData->Abilities)
94 {
95 // Initialize Abilities
96 if (GA != nullptr)
97 {
98 FGameplayAbilitySpec newAbilitySpec{GA.LoadSynchronous(), abilityLevel, inputID++};
99 FGameplayAbilitySpecHandle newAbilitySpecHandle = AbilitySystemComponent->GiveAbility(newAbilitySpec);
100 }
101 }
102 }
103}

105void UARPawnInitComponent::InitializeChargeAttack(UARChargeAttackComponent* InCAC)
106{
107 check(InCAC != nullptr);
108

111 return;
