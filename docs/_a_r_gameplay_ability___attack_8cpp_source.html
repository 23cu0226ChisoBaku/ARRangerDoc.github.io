18void UARGameplayAbility_Attack::GANotify_ImpactResult(USkeletalMeshComponent* MeshComp, UAnimSequenceBase* Animation, const TArray<FGANotify_ImpactResult>& InImpactResults)
19{
20 AActor* avatarActor = GetAvatarActorFromActorInfo();
21 check(avatarActor != nullptr);
22 if (avatarActor == nullptr)
23 {
24 return;
25 }
26
27 for (const FGANotify_ImpactResult& result : InImpactResults)
28 {
29
30 IARAttackable* attackable = ::Cast<IARAttackable>(result.HitActor);
31 if (attackable == nullptr)
32 {
33 continue;
34 }
35
36 FARAttackParameters attackParam{};
37 // TODO Use Avatar location to knockback Target Temporary
38 FVector knockbackDir = result.HitActor->GetActorLocation() - result.ImpactLocation;
39 // Make it Z to zero so we can only use Direction on XY-Plane to determine knockback Direction
40
41 attackParam.Instigator = result.SourceActor;
42 // TODO Stock damage in GA maybe not a great idea
43 attackParam.Damage = AttackDamage;
44 attackParam.bUseAttackerActor = true;
45 attackParam.ImpactLocation = result.ImpactLocation;
46 // attackParam.LaunchDirection = knockbackDir.GetSafeNormal();
47
48 // Try using SourceActor as attacker
49 IARAttackerInterface* attacker = ::Cast<IARAttackerInterface>(result.SourceActor);
50
51 // If SourceActor is not attacker, use avatar
52 if (attacker == nullptr)
53 {
54 attacker = ::Cast<IARAttackerInterface>(avatarActor);
55 }
56
57 // If attacker is still null, try use OccurrenceComp
58 if (attacker == nullptr)
59 {
60 attacker = :: Cast<IARAttackerInterface>(result.OccurrenceComp);
61 }
62
63 FVector knockbackDirNorm = knockbackDir.GetSafeNormal();
64 // Use KnockbackRange to calculate final launch direction
65 if (bClampKnockbackAngle)
66 {
67 // Calculate LaunchDir's projection vector on Plane that attacker's up vector is normal vector
68 FVector norm{ForceInitToZero};
69 FVector attackerFwdDir{ForceInitToZero};
70 if ((attacker != nullptr) && (attacker->GetActor() != nullptr))
71 {
72 norm = attacker->GetActor()->GetActorUpVector();
73 attackerFwdDir = attacker->GetActor()->GetActorForwardVector();
74 }
75 // Fallback to SourceActor
76 else if (result.SourceActor != nullptr)
77 {
78 norm = result.SourceActor->GetActorUpVector();
79 attackerFwdDir = result.SourceActor->GetActorForwardVector();
80 }
81 // Fallback to OccurrenceComp
82 else if (result.OccurrenceComp != nullptr)
83 {
84 norm = result.OccurrenceComp->GetUpVector();
85 attackerFwdDir = result.OccurrenceComp->GetForwardVector();
86 }
87 // Finally, Fallback to avatar
88 else
89 {
90 norm = avatarActor->GetActorUpVector();
91 attackerFwdDir = avatarActor->GetActorForwardVector();
92 }
93
94 norm.Normalize();
95 attackerFwdDir.Normalize();
96
97 const float projectionAngleCos = FVector::DotProduct(norm, knockbackDirNorm);
98 const FVector projectionVec = knockbackDir - (projectionAngleCos * knockbackDir.Length()) * norm;
99 const float curtKnockbackRangeCos = FVector::DotProduct(attackerFwdDir, projectionVec.GetSafeNormal());
100 float curtKnockbackRangeAngleDeg = FMath::RadiansToDegrees(FMath::Acos(curtKnockbackRangeCos));
101 // Clamp current radian
102 curtKnockbackRangeAngleDeg = FMath::Clamp(curtKnockbackRangeAngleDeg, 0.0f, KnockbackAngleHalfRange);
103 // Change knockbackDirNorm
104 if (!FMath::IsNearlyEqual(FMath::Cos(curtKnockbackRangeAngleDeg), curtKnockbackRangeCos))
105 {
106 const float rotateAngleSign = FVector::DotProduct(FVector::CrossProduct(attackerFwdDir, projectionVec).GetSafeNormal(), norm);
107 knockbackDirNorm = attackerFwdDir.RotateAngleAxis(FMath::Sign(rotateAngleSign) * curtKnockbackRangeAngleDeg, norm).GetSafeNormal();
108 }
109 }
110
111 // Finally we put knockbackDirNorm to attackParam
112 attackParam.LaunchDirection = knockbackDirNorm;
113
114 // Apply Attack
115 attackable->AttackTarget(attacker, attackParam);
116 }
117}
