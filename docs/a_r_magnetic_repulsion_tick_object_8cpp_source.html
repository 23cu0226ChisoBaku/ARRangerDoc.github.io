19void UARMagneticRepulsionTickObject::OnTick(const FARPhysicsTickParameters& TickParams, FARPhysicsEvaluationResult& Result)
20{
21 check(Target != nullptr);
22 const AActor* targetActor = Target->GetActor();
23 if (targetActor == nullptr)
24 {
25 return;
26 }
27
28 for (const auto& magnetizedObject : AffectedMagnetizedObjects)
29 {
30 if ((magnetizedObject == nullptr) || (magnetizedObject->GetActor() == nullptr))
31 {
32 continue;
33 }
34
35 AActor* magnetizedObjectActor = magnetizedObject->GetActor();
36 const FVector impactDir = (targetActor->GetActorLocation() - magnetizedObjectActor->GetActorLocation()).GetUnsafeNormal();
37
38 if (impactDir.Z > TOP_DETECTED_COMPONENT_NORM_Z_THRESHOLD)
39 {
40 const FVector targetVelo = targetActor->GetVelocity();
41 if (targetVelo.SizeSquared() > DOUBLE_KINDA_SMALL_NUMBER)
42 {
43 const FVector forwardDir = targetVelo.GetUnsafeNormal();
44 const FVector launchVelo = forwardDir * 1500.0 + FVector{0.0, 0.0, 1200.0};
45 Result.ForceResult += launchVelo;
46 }
47 else
48 {
49 const FVector launchVelo = FVector{0.0, 0.0, 1400.0f};
50 Result.ForceResult += launchVelo;
51 }
52 }
53 else
54 {
55 const FVector horizontalDir = FVector{impactDir.X, impactDir.Y, 0.0}.GetUnsafeNormal();
56 const FVector knockBackVelo = horizontalDir * 800.0f + FVector{0.0, 0.0, 200.0};
57 Result.ForceResult += knockBackVelo;
58 }
59 }
60
61}
