49void UARPhysicsTickObject::Tick(const FARPhysicsTickParameters& TickParams)
50{
51 check(m_internalData.IsValid());
52 if (!m_internalData->bIsEvaluateFinishedCurrentFrame)
53 {
54 FARPhysicsEvaluationResult result{};
55 OnTick(TickParams, result);
56
57 // TODO For blueprint usage
58 // Same as AActor::Tick
59 if (GetClass()->HasAnyClassFlags(CLASS_CompiledFromBlueprint) || !GetClass()->HasAllClassFlags(CLASS_Native))
60 {
61 TickOnBlueprint(TickParams.DeltaTime, TickParams.TotalSimTime, result);
62 }
63
64 EvaluatedResult = result;
65 m_internalData->bIsEvaluateFinishedCurrentFrame = true;
