41void UARMagneticParamEditPanel::NativePreConstruct()
42{
43 Super::NativePreConstruct();
44
45 // Add delegates when this EUW creates outside PIE
46 {
47 if ((GEditor != nullptr) && !GEditor->IsPlayingSessionInEditor())

50 {

58

65 TargetAssetPathDetailsView->OnPropertyChanged_Advanced.AddUniqueDynamic(this, &UARMagneticParamEditPanel::OnTargetFilePathPropertyChanged);
66 }

69 if (TestButton != nullptr)

71 TestButton->OnClicked.AddUniqueDynamic(this, &UARMagneticParamEditPanel::OnAssetPathSaved);

181bool UARMagneticParamEditPanel::TryInitRepository()
182{
183 if (m_repositoryConnection == nullptr)
184 {
185 return false;
186 }
187
188 UObject* assetObj = UARRepositoryAssetHelper::GetAssetByClass
189 (
190 TargetAssetPath.FilePath, // Long package name
191 UMagneticParametersList::StaticClass(),
192 nullptr,
193 false // Find asset without creating one
194 );
195
196 if (assetObj == nullptr)
197 {
198 UE_LOG(LogTemp, Error, TEXT("Load failed. File path: [%s]"), *TargetAssetPath.FilePath);
199 return false;
200 }
201
202 if (UMagneticParametersList* actualAsset = Cast<UMagneticParametersList>(assetObj))
203 {
204 for (const auto& [ name, dataEntry ] : actualAsset->MagneticParameters)
205 {
206 ARRanger::Physics::FMagneticParameterDTO dto = FMagneticParameterDTO {dataEntry.MagneticCharge, dataEntry.MegneticObjectMass};
207
208 if (!m_repositoryConnection->Exist(dataEntry.MagneticPhysicsUserClass->GetDefaultObject()))
209 {
210 m_repositoryConnection->Add(dataEntry.MagneticPhysicsUserClass->GetDefaultObject(), dto);
211 }
212 else
213 {
214 m_repositoryConnection->Save(dataEntry.MagneticPhysicsUserClass->GetDefaultObject(), dto);
215 }
216 }
217
218 return true;
219 }
220 else
221 {
222 UE_LOG(LogTemp, Error, TEXT("Asset type cast failed.Current asset type:[%s]"), *assetObj->GetClass()->GetName());
223 }
224
225 return false;
226}

244void UARMagneticParamEditPanel::OnAssetPathSaved()
245{
246 UFactory* assetPathSaveFactory = NewObject<UMagneticParamAssetSavePathFactory>();
247
248 const FString saveFileLongPackageName = FPackageName::FilenameToLongPackageName(FARMagneticParamEditHelper::MagneticParamAssetPathSaveFileAbsolutePath);
249 UObject* assetObj = UARRepositoryAssetHelper::GetAssetByClass
250 (
251 saveFileLongPackageName,
252 UMagneticParamAssetPathSave::StaticClass(),
253 assetPathSaveFactory
254 );
255
256 if (assetObj == nullptr)
257 {
258 UE_LOG(LogTemp, Error, TEXT("Save failed"));
259 return;
260 }
261
262 if (UMagneticParamAssetPathSave* actualAsset = Cast<UMagneticParamAssetPathSave>(assetObj))
263 {
264 // FilePath was saved as long package name before
265 actualAsset->Save(ConvertLongPackageNameToObjectPath(TargetAssetPath.FilePath));
266
267 // Save asset after modifying file
268 UARRepositoryAssetHelper::SaveMagneticParametersAsset(saveFileLongPackageName);
269 }

285 FString ConvertAbsolutePathToObjectPath(const FString& InPath)
286 {
287 FString relativePath = InPath;
288
289 const FString contentDir = FPaths::ConvertRelativePathToFull(FPaths::ProjectContentDir());
290 if (InPath.StartsWith(contentDir))
291 {
292 relativePath.Reset();
293
294 const FString relativePathWithExt = InPath.RightChop(contentDir.Len());
295 const FString baseFileName = FPaths::GetBaseFilename(InPath);
296 // Remove extension
297 relativePath.Append(FPaths::ChangeExtension(relativePathWithExt, TEXT("")));
298 relativePath = FString::Printf(TEXT("/Game/%s.%s"), *relativePath, *baseFileName);
299 }
300
301 return relativePath;
302 }
303
304 FString ConvertLongPackageNameToObjectPath(const FString& InPackageName)
305 {
306 if (FPackageName::IsValidLongPackageName(InPackageName))
307 {
308 const FString absolutePath = FPaths::ConvertRelativePathToFull(FPackageName::LongPackageNameToFilename(InPackageName));
309 return ConvertAbsolutePathToObjectPath(absolutePath);
310 }
311
312 return FString{};
313 }
