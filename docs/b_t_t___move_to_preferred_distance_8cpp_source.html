14EBTNodeResult::Type UBTT_MoveToPreferredDistance::ExecuteTask(UBehaviorTreeComponent& OwnerComp, uint8* NodeMemory)
15{
16 UBlackboardComponent* BB = OwnerComp.GetBlackboardComponent();
17 if (!BB) return EBTNodeResult::Failed;
18
19 AActor* Target = Cast<AActor>(BB->GetValueAsObject("TargetActor"));
20 APawn* ControlledPawn = OwnerComp.GetAIOwner()->GetPawn();
21
22 if (!Target || !ControlledPawn) return EBTNodeResult::Failed;
23
24 float PreferredDistance = BB->GetValueAsFloat("PreferredDistance");
25 FVector ToTarget = ControlledPawn->GetActorLocation() - Target->GetActorLocation();
26 float CurrentDistance = ToTarget.Size();
27
28 // すでに距離が近いなら移動不要
29 if (FMath::Abs(CurrentDistance - PreferredDistance) < 100.f)
30 {
31 return EBTNodeResult::Succeeded;
32 }
33
34 ToTarget.Normalize();
35 FVector DesiredLocation = Target->GetActorLocation() + ToTarget * PreferredDistance;
36
37 // NavMesh補正
38 UNavigationSystemV1* NavSys = UNavigationSystemV1::GetCurrent(GetWorld());
39 FNavLocation Projected;
40 if (NavSys && NavSys->ProjectPointToNavigation(DesiredLocation, Projected))
41 {
42 CurrentMoveGoal = Projected.Location;
43 OwnerComp.GetAIOwner()->MoveToLocation(CurrentMoveGoal, 50.0f); // 半径50で到達判定
44 bIsMoving = true;
45 return EBTNodeResult::InProgress;
46 }
47
48 return EBTNodeResult::Failed;
49}

51void UBTT_MoveToPreferredDistance::TickTask(UBehaviorTreeComponent& OwnerComp, uint8* NodeMemory, float DeltaSeconds)
52{
53 AAIController* AICon = OwnerComp.GetAIOwner();

58 FinishLatentTask(OwnerComp, EBTNodeResult::Failed);

62 AActor* Target = Cast<AActor>(BB->GetValueAsObject("TargetActor"));
63 if (!Target)
64 {
65 FinishLatentTask(OwnerComp, EBTNodeResult::Failed);
66 return;

68
69 float PreferredDistance = BB->GetValueAsFloat("PreferredDistance");
70 FVector ToTarget = ControlledPawn->GetActorLocation() - Target->GetActorLocation();
71 float CurrentDistance = ToTarget.Size();
72
73 // 目標距離に収まったらタスク終了
74 if (FMath::Abs(CurrentDistance - PreferredDistance) < 100.f)
75 {
76 AICon->StopMovement();
77 FinishLatentTask(OwnerComp, EBTNodeResult::Succeeded);
78 return;
79 }
80
81 // ゴールを再計算
82 ToTarget.Normalize();
83 FVector DesiredLocation = Target->GetActorLocation() + ToTarget * PreferredDistance;
84
85 UNavigationSystemV1* NavSys = UNavigationSystemV1::GetCurrent(GetWorld());
86 FNavLocation Projected;
87 if (NavSys && NavSys->ProjectPointToNavigation(DesiredLocation, Projected))
88 {
89 AICon->MoveToLocation(Projected.Location, 50.f, false);
90 }
