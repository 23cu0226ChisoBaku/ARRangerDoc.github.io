97 int32 FRangeDetector::Evaluate(const FRangeDetectorEvaluationParameter& EvaluationParam)
98 {
99 if (!IsActivate())
100 {
101 UE_LOG(LogARRangeDetector, Warning, TEXT("Range detector is not activated"));
102 return 0;
103 }
104 else if (!m_constData.IsValid())
105 {
106 UE_LOG(LogARRangeDetector, Error, TEXT("Range detector is invalid.Remove it immediately"));
107 return 0;
108 }
109
110 m_evaluatedResult.Reset();
111
112 if (EvaluationParam.OriginSceneComp != nullptr)
113 {
114 TArray<AActor*> detectResults{};
115 const FVector originLocation = EvaluationParam.OriginSceneComp->GetComponentLocation();
116 const FRotator originRotation = EvaluationParam.OriginSceneComp->GetComponentRotation();
117 const FVector originScale3D = EvaluationParam.OriginSceneComp->GetComponentScale();
118
119 (void)m_constData->DetectTargets(
120 EvaluationParam.World,
121 EvaluationParam.OriginActor,
122 originLocation,
123 originRotation,
124 originScale3D,
125 detectResults
126 );
127
128 m_evaluatedResult.DetectedActors.Append(detectResults);
129 }
130
131 CheckFilterValidation();
132 FilterResult();
133
134 return m_evaluatedResult.GetResultNum();
135 }
