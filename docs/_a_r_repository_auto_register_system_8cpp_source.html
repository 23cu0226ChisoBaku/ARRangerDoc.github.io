18void UARRepositoryAutoRegisterSystem::Initialize(FSubsystemCollectionBase& Collection)
19{
20 Super::Initialize(Collection);
21
22 if (UARGameInstance* gameInstance = Cast<UARGameInstance>(GetGameInstance()))
23 {
24 // Initialize MagneticParam Repository
25 if (!gameInstance->MagneticParamsAsset.IsNull())
26 {
27 FARPhysicsMagneticParametersRepository* magRepo = nullptr;
28 if (FARRepositoryManager::Get().HasRepositoryRegistered(FARPhysicsMagneticParametersRepository::GetStaticRepositoryName()))
29 {
30 magRepo = FARRepositoryManager::Get().GetRepository<FARPhysicsMagneticParametersRepository>();
31 }
32 else
33 {
34 magRepo = FARRepositoryManager::Get().RegisterRepository<FARPhysicsMagneticParametersRepository>();
35 }
36
37 if (magRepo != nullptr)
38 {
39 UMagneticParametersList* magneticAsset = gameInstance->MagneticParamsAsset.LoadSynchronous();
40 for (const auto& [ dataName , data ] : magneticAsset->MagneticParameters)
41 {
42 if (data.MagneticPhysicsUserClass == nullptr)
43 {
44 continue;
45 }
46
47 UE_LOG(LogTemp, Warning, TEXT("Loading mag data : Name[%s]"), *dataName);
48 FMagneticParameterDTO dto =
49 {
50 .MagneticCharge = data.MagneticCharge,
51 .MagneticObjectMass = data.MegneticObjectMass
52 };
53
54
55 if (!magRepo->Exist(data.MagneticPhysicsUserClass->GetDefaultObject()))
56 {
57 const bool bAddSuccess = magRepo->Add(data.MagneticPhysicsUserClass->GetDefaultObject(), dto);
58 if (!bAddSuccess)
59 {
60 UE_LOG(LogTemp, Error, TEXT("Data add failed. [%s]"), *dataName);
61 }
62 }
63 else
64 {
65 const bool bSaveSuccess = magRepo->Save(data.MagneticPhysicsUserClass->GetDefaultObject(), dto);
66 if (!bSaveSuccess)
67 {
68 UE_LOG(LogTemp, Error, TEXT("Data save failed. [%s]"), *dataName);
69 }
70 }
71
72 }
73 }
74
75#if WITH_EDITORONLY_DATA
76 if (OnRepositoryRegistered.IsBound())
77 {
78 OnRepositoryRegistered.Broadcast();
79 }
80#endif
81 }
82 }
83}
